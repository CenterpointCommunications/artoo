Example: hello_api

require 'artoo/robot'

class HelloRobot < Artoo::Robot
  connection :loop
  device :passthru

  api :host => '127.0.0.1', :port => '4321'

  work do
          puts "Hello from the API running at \1api_host}:\1api_port}..."
  end

  def hello name
    "hi \1name}!"
  end
end

robots = []
robots << HelloRobot.new(:name => "hello", :commands => [:hello])
HelloRobot.work!(robots)
